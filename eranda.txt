#include <stdio.h>

double VolumeOfCone(double radius, double height);
double VOlumeOfCylinder(double radius, double height);

int main(void) {
  double height = 200.0; // declare height variable

  printf("Radius\t\tVolume of code\t\tVolume of cylinder\n\n");
  int i;
  for (i = 50; i <= 150; i += 10) {
    double r = i / 1.0;
    // print output
    printf("%.0lf\t\t\t%.2lf\t\t\t%.2lf\n", r, VolumeOfCone(r, height),
           VOlumeOfCylinder(r, height));
  }

  return 0;
}
// calculate volume in cone
double VolumeOfCone(double radius, double height) {
  return (1 / 3.0) * (22 / 7.0) * radius * radius * height;
}
// calculate volume in cylinder
double VOlumeOfCylinder(double radius, double height) {
  return (22 / 7.0) * radius * radius * height;
}
